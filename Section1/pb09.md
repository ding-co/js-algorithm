## 9. A를 #으로

### _문제 풀이_

```html
<!-- My Solution -->

<html>
  <head>
    <meta charset="UTF-8" />
    <title>A를 #으로</title>
  </head>
  <body>
    <script>
      function solution(s) {
        let answer = '';
        for (let element of s) {
          if (element === 'A') {
            element = '#';
          }
          answer += element;
        }
        return answer;
      }

      let str = 'BANANA';
      console.log(solution(str));
    </script>
  </body>
</html>
```

```html
<!-- Lecture Solution 1  -->

<html>
  <head>
    <meta charset="UTF-8" />
    <title>A를 #으로</title>
  </head>
  <body>
    <script>
      function solution(s) {
        let answer = '';
        for (let element of s) {
          if (element === 'A') answer += '#';
          else answer += element;
        }
        return answer;
      }

      let str = 'BANANA';
      console.log(solution(str));
    </script>
  </body>
</html>
```

```html
<!-- Lecture Solution 2 -->

<html>
  <head>
    <meta charset="UTF-8" />
    <title>출력결과</title>
  </head>
  <body>
    <script>
      function solution(s) {
        // 문자열은 값이 복사됨 (주소가 참조되는 개념이 아님)
        let answer = s;

        // A라는 대문자를 찾아서 #으로 바꿔라
        // g를 붙이면 global 로 다 바꿈
        answer = answer.replace(/A/g, '#');

        return answer;
      }

      let str = 'BANANA';
      console.log(solution(str));
    </script>
  </body>
</html>
```
