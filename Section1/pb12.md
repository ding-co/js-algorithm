```html
<!-- Lecture Solution 1 -->

<html>
  <head>
    <meta charset="UTF-8" />
    <title>출력결과</title>
  </head>
  <body>
    <script>
      function solution(s) {
        let answer = '';
        for (let x of s) {
          if (x === x.toLowerCase()) answer += x.toUpperCase();
          else answer += x;
        }
        return answer;
      }

      let str = 'ItisTimeToStudy';
      console.log(solution(str));
    </script>
  </body>
</html>
```

    <script>
      function solution(s) {
        let answer = '';
        for (let x of s) {
          let num = x.charCodeAt();
          if (num >= 97 && num <= 122) answer += String.fromCharCode(num - 32);
          else answer += x;
        }
        return answer;
      }

      let str = 'ItisTimeToStudy';
      console.log(solution(str));
    </script>
  </body>
</html>
