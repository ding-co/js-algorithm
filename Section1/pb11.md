## 11. 대문자 찾기

### _문제 풀이_

```html
<!-- My Solution -->

<html>
  <head>
    <meta charset="UTF-8" />
    <title>출력결과</title>
  </head>
  <body>
    <script>
      function solution(s) {
        let answer = 0;
        for (let element of s) {
          if (element === element.toUpperCase()) answer += 1;
        }
        return answer;
      }

      let str = 'KoreaTimeGood';
      console.log(solution(str));
    </script>
  </body>
</html>
```

```html
<!-- Lecture Solution 2 -->

<html>
  <head>
    <meta charset="UTF-8" />
    <title>출력결과</title>
  </head>
  <body>
    <script>
      function solution(s) {
        let answer = 0;
        for (let x of s) {
          // 아스키코드 값
          // 대문자: 65 ~ 90 (A ~ Z)
          // 소문자: 97 ~ 122 (a ~ z)
          let num = x.charCodeAt();
          if (num >= 65 && num <= 90) answer++;
        }

        return answer;
      }

      let str = 'KoreaTimeGood';
      console.log(solution(str));
    </script>
  </body>
</html>
```
