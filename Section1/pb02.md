## 1. 삼각형 판별하기

### _문제 풀이_

- 삼각형 되기 위한 조건
- 가장 긴 막대 길이 < 나머지 두 막대 길이 합

```html
<!-- My Solution -->

<html>
  <head>
    <meta charset="UTF-8" />
    <title>삼각형 판별</title>
  </head>
  <body>
    <script>
      function solution(a, b, c) {
        let answer = 'YES';
        let max;

        if (a > b && a > c) {
          max = a;
          if (max >= b + c) {
            answer = 'NO';
          }
        } else if (b > c) {
          max = b;
          if (max >= a + c) {
            answer = 'NO';
          }
        } else {
          max = c;
          if (max >= a + b) {
            answer = 'NO';
          }
        }
        return answer;
      }

      console.log(solution(13, 33, 17));
    </script>
  </body>
</html>
```

```html
<!-- Lecture Solution  -->

<html>
  <head>
    <meta charset="UTF-8" />
    <title>삼각형 판별</title>
  </head>
  <body>
    <script>
      function solution(a, b, c) {
        let answer = 'YES',
          max;
        let sum = a + b + c;

        if (a > b) max = a;
        else max = b;
        if (c > max) max = c;

        if (max >= sum - max) answer = 'NO';

        return answer;
      }

      console.log(solution(13, 33, 17));
    </script>
  </body>
</html>
```
