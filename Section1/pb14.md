## 14. 가장 긴 문자열

### _문제 풀이_

```html
<!-- My Solution -->

<html>
  <head>
    <meta charset="UTF-8" />
    <title>출력결과</title>
  </head>
  <body>
    <script>
      function solution(s) {
        let answer;
        let count = 0;
        for (let i = 0; i < s.length; i++) {
          if (s[i].length > count) {
            count = s[i].length;
            answer = s[i];
          }
        }
        return answer;
      }
      let str = ['teacher', 'time', 'student', 'beautiful', 'good'];
      console.log(solution(str));
    </script>
  </body>
</html>
```

      function solution(s) {
        let answer,
          max = Number.MIN_SAFE_INTEGER;
        for (let x of s) {
          if (x.length > max) {
            max = x.length;
            answer = x;
          }
        }
        return answer;
      }
      let str = ['teacher', 'time', 'student', 'beautiful', 'good'];
      console.log(solution(str));
    </script>
  </body>
</html>
