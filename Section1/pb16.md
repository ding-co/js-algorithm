## 16.중복 문자 제거

### _문제 풀이_

```html
<!-- Lecture Solution -->

<html>
  <head>
    <meta charset="UTF-8" />
    <title>출력결과</title>
  </head>
  <body>
    <script>
      function solution(s) {
        let answer = '';
        // indexOf 메서드: element 첫번째로 발견되는 인덱스 위치 return
        // indexOf(element, index), index 부터 첫 발견 위치
        // element 발견 못하면 -1 return

        for (let i = 0; i < s.length; i++) {
          if (s.indexOf(s[i]) === i) answer += s[i];
        }
        return answer;
      }
      console.log(solution('ksekkset'));
    </script>
  </body>
</html>
```

```html
<!-- Reference -->

<html>
  <head>
    <meta charset="UTF-8" />
    <title>출력결과</title>
  </head>
  <body>
    <script>
      function solution(s) {
        // 특정 문자 개수 찾기 (번외)
        let answer = 0;
        let pos = s.indexOf('k');
        while (pos !== -1) {
          answer++;
          pos = s.indexOf('k', pos + 1);
        }
        return answer;
      }
      console.log(solution('ksekkset'));
    </script>
  </body>
</html>
```
