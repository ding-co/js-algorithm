## 8. 일곱 난쟁이

### _문제 풀이_

- 2중 중첩 for문
  - i, j (가짜 난쟁이 인덱스 찾기)
- (sum - (arr[i] + arr[j])) === 100
- arr.splice(x, 1)

```html
<!-- My Solution -->

<html>
  <head>
    <meta charset="UTF-8" />
    <title>일곱 난쟁이</title>
  </head>
  <body>
    <script>
      function solution(arr) {
        let answer;
        let sum = 0;

        for (let l = 0; l < arr.length; l++) {
          sum += arr[l];
        }

        for (let i = 0; i < arr.length - 1; i++) {
          for (let j = i + 1; j < arr.length; j++) {
            if (sum - (arr[i] + arr[j]) === 100) {
              arr.splice(i, 1);
              arr.splice(j - 1, 1);
            }
          }
        }

        answer = arr;

        return answer;
      }

      let arr = [20, 7, 23, 19, 10, 15, 25, 8, 13];
      console.log(solution(arr));
    </script>
  </body>
</html>
```

<!-- Lecture Solution  -->

<html>
  <head>
    <meta charset="UTF-8" />
    <title>일곱 난쟁이</title>
  </head>
  <body>
    <script>
      function solution(arr) {
        // 얕은 복사 (같은 객체 참조)
        let answer = arr;

        // a에 계속 누적 (accu), b는 배열
        // a를 0으로 초기화
        let sum = arr.reduce((a, b) => a + b, 0);
        for (let i = 0; i < 8; i++) {
          for (let j = i + 1; j < 9; j++) {
            if (sum - (arr[i] + arr[j]) === 100) {
              arr.splice(j, 1);
              arr.splice(i, 1);
            }
          }
        }

        return answer;
      }

      let arr = [20, 7, 23, 19, 10, 15, 25, 8, 13];
      console.log(solution(arr));
    </script>
  </body>
</html>
