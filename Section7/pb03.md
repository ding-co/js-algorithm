## 3. Special Sort (버블정렬 응용)

### _문제 풀이_

- 버블정렬 이용
  - 앞이 양수, 뒤가 음수이면 스왑

```html
<!-- My Solution 01 -->
<html>
  <head>
    <meta charset="UTF-8" />
    <title>출력결과</title>
  </head>
  <body>
    <script>
      function solution(arr) {
        let answer = arr;
        const minusArray = answer.filter((num) => num < 0);
        const plusArray = answer.filter((num) => num > 0);
        answer = minusArray.concat(plusArray);
        return answer;
      }

      let arr = [1, 2, 3, -3, -2, 5, 6, -6];
      console.log(solution(arr));
    </script>
  </body>
</html>
```

```html
<!-- My Solution 02 -->
<html>
  <head>
    <meta charset="UTF-8" />
    <title>출력결과</title>
  </head>
  <body>
    <script>
      function solution(arr) {
        let answer = arr;
        for (let i = 0; i < answer.length - 1; i += 1) {
          for (let j = 0; j < answer.length - i - 1; j += 1) {
            if (answer[j] > 0 && answer[j + 1] < 0)
              [answer[j], answer[j + 1]] = [answer[j + 1], answer[j]];
          }
        }
        return answer;
      }

      let arr = [1, 2, 3, -3, -2, 5, 6, -6];
      console.log(solution(arr));
    </script>
  </body>
</html>
```

        let answer = arr;
        for (let i = answer.length - 1; i >= 1; i -= 1) {
          for (let j = 0; j <= i; j += 1) {
            if (answer[j] > 0 && answer[j + 1] < 0)
              [answer[j], answer[j + 1]] = [answer[j + 1], answer[j]];
          }
        }
        return answer;
      }
