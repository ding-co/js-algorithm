## 1. 자릿수의 합

### _문제 풀이_

- Number 그대로 이용
  - 10으로 나눈 나머지 => 일의 자리
  - 10으로 나눈 몫에 10으로 나눈 나머지 => 십의 자리
  - 10으로 나눈 몫에 10으로 나눈 몫에 10으로 나눈 나머지 => 백의 자리
- String으로 변환해서 이용

        let answer = 0;
        let max = Number.MIN_SAFE_INTEGER;
        for (let x of arr) {
          x = String(x);
          let sum = 0;
          for (let y of x) {
            sum += Number(y);
          }
          if (sum >= max) {
            if (answer < x) answer = x;
            max = sum;
          }
        }
        return answer;
      }

      function solution(n, arr) {
        let answer;
        let max = Number.MIN_SAFE_INTEGER;
        for (let x of arr) {
          let sum = 0;
          let tmp = x;
          while (tmp) {
            sum += tmp % 10;
            tmp = Math.floor(tmp / 10);
          }
          if (sum > max) {
            max = sum;
            answer = x;
          } else if (sum === max) {
            if (answer < x) answer = x;
          }
        }
        return answer;
      }
      let arr = [128, 460, 603, 40, 521, 137, 123];
      console.log(solution(7, arr));
    </script>

      function solution(n, arr) {
        let answer;
        let max = Number.MIN_SAFE_INTEGER;
        for (let x of arr) {
          let sum = x
            .toString()
            .split('')
            .reduce((a, b) => a + Number(b), 0);
          if (sum > max) {
            max = sum;
            answer = x;
          } else if (sum === max) {
            if (answer < x) answer = x;
          }
        }
        return answer;
      }

      console.log(solution(7, arr));
