## 4. 가장 짧은 문자거리

- 왼쪽 타겟으로부터 떨어진 거리
- 오른쪽 타겟으로부터 떨어진 거리

### _문제 풀이_

```html
<!-- Lecture Solution -->

<html>
  <head>
    <meta charset="UTF-8" />
    <title>출력결과</title>
  </head>
  <body>
    <script>
      function solution(s, t) {
        let answer = [];
        let p = 1000;
        for (let x of s) {
          if (x === t) {
            p = 0;
            answer.push(p);
          } else {
            p++;
            answer.push(p);
          }
        }
        p = 1000;
        for (let i = s.length - 1; i >= 0; i--) {
          if (s[i] === t) p = 0;
          else {
            p++;
            answer[i] = Math.min(answer[i], p);
          }
        }
        return answer;
      }

      let str = 'teachermode';
      console.log(solution(str, 'e'));
    </script>
  </body>
</html>
```
