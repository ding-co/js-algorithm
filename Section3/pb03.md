## 3. 숫자만 추출

### _문제 풀이_

```html
<!-- My Solution -->

<html>
  <head>
    <meta charset="UTF-8" />
    <title>출력결과</title>
  </head>
  <body>
    <script>
      function solution(str) {
        let answer = '';
        str = str.toLowerCase();
        answer = str.replace(/[a-z]/g, '');
        return Number(answer);
      }
      let str = 'g0en2T0s8eSoft';
      console.log(solution(str));
    </script>
  </body>
</html>
```

```html
<!-- Lecture Solution 1 -->

<html>
  <head>
    <meta charset="UTF-8" />
    <title>출력결과</title>
  </head>
  <body>
    <script>
      function solution(str) {
        let answer = '';
        for (let x of str) {
          if (!isNaN(x)) answer += x;
        }
        return parseInt(answer);
      }
      let str = 'g0en2T0s8eSoft';
      console.log(solution(str));
    </script>
  </body>
</html>
```

        let answer = 0;

        }
        return answer;
      }
      let str = 'g0en2T0s8eSoft';
