```html
    <script>
      function solution(s) {
        let answer = 'YES';
        s = s.toLowerCase();
        let len = s.length;
        for (let i = 0; i < Math.floor(len / 2); i++) {
          if (s[i] !== s[len - i - 1]) return 'NO';
        }
        return answer;
      }

      let str = 'gooG';
      console.log(solution(str));
    </script>
  </body>
</html>
```

```html
      function solution(s) {
        let answer = 'YES';
        s = s.toLowerCase();
        // reverse는 배열의 method
        if (s !== s.split('').reverse().join('')) return 'NO';
        return answer;
      }

      let str = 'gooG';
      console.log(solution(str));
    </script>
```
