## 1. 회문 문자열

### _문제 풀이_

```html
<!-- My Solution -->

<html>
  <head>
    <meta charset="UTF-8" />
    <title>출력결과</title>
  </head>
  <body>
    <script>
      function solution(s) {
        let answer = 'YES';
        s = s.toLowerCase().split('');
        while (s.length > 1) {
          if (s.shift() !== s.pop()) {
            answer = 'NO';
            break;
          }
        }
        return answer;
      }
      let str = 'gooG';
      console.log(solution(str));
    </script>
  </body>
</html>
```

```html
<!-- Lecture Solution 1 -->

<html>
  <head>
    <meta charset="UTF-8" />
    <title>출력결과</title>
  </head>
  <body>
    <script>
      function solution(s) {
        let answer = 'YES';
        s = s.toLowerCase();
        let len = s.length;
        for (let i = 0; Math.floor(i < len / 2); i++) {
          if (s[i] !== s[len - i - 1]) return 'NO';
        }
        return answer;
      }
      let str = 'gooG';
      console.log(solution(str));
    </script>
  </body>
</html>
```
