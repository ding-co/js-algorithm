## 2. 유효한 팰린드롬

### _문제 풀이_

```html
<!-- My Solution -->

<html>
  <head>
    <meta charset="UTF-8" />
    <title>출력결과</title>
  </head>
  <body>
    <script>
      function solution(s) {
        let answer = 'YES';
        s = s.toLowerCase();
        s = s.split('').filter((value) => value >= 'a' && value <= 'z');
        if (s.join('') !== s.reverse().join('')) return 'NO';
        return answer;
      }

      let str = 'found7, time: study; Yduts; emit, 7Dnuof';
      console.log(solution(str));
    </script>
  </body>
</html>
```

```html
<!-- Lecture Solution -->

<html>
  <head>
    <meta charset="UTF-8" />
    <title>출력결과</title>
  </head>
  <body>
    <script>
      function solution(s) {
        let answer = 'YES';
        s = s.toLowerCase().replace(/[^a-z]/g, '');
        if (s !== s.split('').reverse().join('')) return 'NO';
        return answer;
      }

      let str = 'found7, time: study; Yduts; emit, 7Dnuof';
      console.log(solution(str));
    </script>
  </body>
</html>
```
