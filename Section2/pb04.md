## 4. 점수 계산

### _문제 풀이_

```html
<!-- My Solution -->

<html>
  <head>
    <meta charset="UTF-8" />
    <title>출력결과</title>
  </head>
  <body>
    <script>
      function solution(arr) {
        let answer = 0;
        let score = 0;
        for (let i = 0; i < arr.length; i++) {
          if (arr[i] === 1) score += 1;
          answer += score;
          if (arr[i + 1] === 0) score = 0;
        }
        return answer;
      }

      let arr = [1, 0, 1, 1, 1, 0, 0, 1, 1, 0];
      console.log(solution(arr));
    </script>
  </body>
</html>
```

```html
<!-- Lecture Solution -->

<html>
  <head>
    <meta charset="UTF-8" />
    <title>출력결과</title>
  </head>
  <body>
    <script>
      function solution(arr) {
        let answer = 0;
        let cnt = 0;
        for (let x of arr) {
          if (x === 1) {
            cnt++;
            answer += cnt;
          } else cnt = 0;
        }
        return answer;
      }

      let arr = [1, 0, 1, 1, 1, 0, 0, 1, 1, 0];
      console.log(solution(arr));
    </script>
  </body>
</html>
```
